    private void jButtonCadastrarServicoActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButtonCadastrarServicoActionPerformed
        try {
            //        if ((jTextFieldnomePeca.getText().trim().isEmpty() && jTextFieldprecoPeca.getText().trim().isEmpty()) == false) {
            //          Peca peca = new Peca();
            //        peca.setDescricao(jTextFieldnomePeca.getText());
            //      peca.setPreco(Float.parseFloat(jTextFieldprecoPeca.getText()));
            //    DadosPeca a = new DadosPeca();
            //jTextFieldnomePeca.setText("");
            //  jTextFieldprecoPeca.setText("");
            // a.cadastrarPeca(peca);
            //     JOptionPane.showMessageDialog(null, "Peça cadastrada com sucesso");
            //       jTextFieldnomePeca.requestFocus();
            //     }
            Servico servico = new Servico();
            Operacao op = new Operacao();
            NegocioServico negocio = new NegocioServico();
            servico.setDescricao(jTextFieldnomeServico.getText());
            String textoPreco = jTextFieldprecoServico.getText();
            if (textoPreco.length() > Float.MAX_VALUE) {
                JOptionPane.showMessageDialog(null, "Preço informado estoura a capacidade de armazenamento da variável.");
                jTextFieldprecoServico.setText("");
                jTextFieldprecoServico.requestFocus();
            }
            Float preco = op.formatadorVirgulaParaPonto(textoPreco);
            servico.setPreco(preco);
            negocio.cadastrarServico(servico);

            JOptionPane.showMessageDialog(null, "Serviço cadastrado com sucesso");
            jTextFieldnomeServico.setText("");
            jTextFieldprecoServico.setText("");
            jTextFieldnomeServico.requestFocus();

        }  catch (NumberFormatException e) {

            // Validação de NumberFormat quando o usuário deixar os 2 campos em branco. 
            // Ele vai entrar no catch por não conseguir converter a variável preco com o método formatadorVirgulaParaPonto, que retorna uma conversão de tipos.
            if (jTextFieldnomeServico.getText().trim().isEmpty()) {
                JOptionPane.showMessageDialog(null, "Informe o nome do servico.");
                jTextFieldnomeServico.requestFocus();
            } else {
                try {
                    Servico servico = new Servico();
                    servico.setDescricao(jTextFieldnomeServico.getText());
                    NegocioServico validaServico = new NegocioServico();
                    validaServico.cadastrarServico(servico);

                } catch (Exception ex) {
                    JOptionPane.showMessageDialog(null, ex.getMessage());
                    jTextFieldprecoServico.setText("");
                    jTextFieldprecoServico.requestFocus();
                }

            }
        } catch (Exception e) {

            JOptionPane.showMessageDialog(null, e.getMessage());
            jTextFieldprecoServico.setText("");
            jTextFieldprecoServico.requestFocus();
        }

    }//GEN-LAST:event_jButtonCadastrarServicoActionPerformed
