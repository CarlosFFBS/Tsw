    @Override

    @Override
    public int consultaCodClientePF(ClientePessoaFisica cliente) throws Exception {
        String sql = " SELECT Cli.cli_Codigo ";
        sql += " FROM CLIENTE AS Cli ";
        sql += " WHERE Cli.email = ? ";
        PreparedStatement stmt = super.conectarParametros().prepareStatement(sql);
        stmt.setString(1, cliente.getEmail());
        ResultSet rs = stmt.executeQuery();
        while (rs.next()) {
            cliente.setCli_Codigo(rs.getInt("cli_Codigo"));
        }
        return cliente.getCli_Codigo();

    }
